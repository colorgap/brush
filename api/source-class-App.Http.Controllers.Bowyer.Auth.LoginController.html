31     /**
32     * validateLogin validates the username/email and password from database.
33     * @param Request
34     * @return Notification
35     */
36     public function validateLogin(Request $request){
37         $user = User::where('email',$request->input("username"))
38             ->orWhere('username',$request->input("username"))
39             ->Where('password',hash('sha1', $request->input("password")))->first();
40         if(!empty($user)){
41             $salt = new Salt();
42             $user->api_token = hash('sha1',$salt->spiceItUp($user->email));
43             $user->save();
44             return $user;
45         }else{
46             $error = new Notification();
47             $error->notify("Provided Username and Password doesn't match. Please try again.",5000);
48             return $error;
49         }
50     }
51     /**
52     * logout method gets user details from the Auth::user and removes api token from db and send back notification. 
53     * @return Notification
54     */
55     public function logout(){
56         $user = Auth::user();
57         if(!empty($user)){
58             $user->api_token = null;
59             $user->save();
60         }
